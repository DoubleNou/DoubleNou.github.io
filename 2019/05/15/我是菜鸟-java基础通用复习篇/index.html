    int left = 0;

    int right = array.length - 1;

    while ( left <= right){

        int mid = (left + right)/2 ;
        if(array[mid].equals(value)){
            return mid;
        }

        if(Integer.valueOf(value) > Integer.valueOf(array[mid])){
            left = mid+1;
        }
        if(Integer.valueOf(value) < Integer.valueOf(array[mid])){
            right = mid-1;
        }
    }
    return  -1;
}
